<script>
  // üîë REPLACE WITH YOUR OPENAI API KEY
  const OPENAI_API_KEY = "YOUR_OPENAI_API_KEY";

  const chatMessages = document.getElementById("chat-messages");
  const chatInput = document.getElementById("chat-input");
  const chatSend = document.getElementById("chat-send");
  const chatVoice = document.getElementById("chat-voice");

  // Send message to ChatGPT
  async function sendMessage(message) {
    if (!message) return;

    // Show user message
    const userMsgDiv = document.createElement("div");
    userMsgDiv.textContent = "You: " + message;
    chatMessages.appendChild(userMsgDiv);
    chatMessages.scrollTop = chatMessages.scrollHeight;

    // Call OpenAI API
    try {
      const response = await fetch("https://api.openai.com/v1/chat/completions", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${OPENAI_API_KEY}`
        },
        body: JSON.stringify({
          model: "gpt-3.5-turbo",
          messages: [{role: "user", content: message}],
          max_tokens: 300
        })
      });
      const data = await response.json();
      const aiText = data.choices[0].message.content;

      // Show AI reply
      const aiMsgDiv = document.createElement("div");
      aiMsgDiv.textContent = "AI: " + aiText;
      aiMsgDiv.style.color = "blue";
      chatMessages.appendChild(aiMsgDiv);
      chatMessages.scrollTop = chatMessages.scrollHeight;

      // Speak AI reply
      if ('speechSynthesis' in window) {
        const utterance = new SpeechSynthesisUtterance(aiText);
        // Auto detect Japanese or English
        utterance.lang = /[\u3040-\u30ff\u31f0-\u31ff]/.test(aiText) ? 'ja-JP' : 'en-US';
        speechSynthesis.speak(utterance);
      }
    } catch (error) {
      alert("Error connecting to OpenAI: " + error.message);
    }
  }

  // Send button
  chatSend.addEventListener("click", () => {
    const message = chatInput.value.trim();
    sendMessage(message);
    chatInput.value = "";
  });

  // Enter key
  chatInput.addEventListener("keypress", (e) => {
    if (e.key === "Enter") chatSend.click();
  });

  // Voice input button (supports English & Japanese)
  chatVoice.addEventListener("click", () => {
    if (!('webkitSpeechRecognition' in window)) {
      alert("Your browser does not support voice input.");
      return;
    }

    const recognition = new webkitSpeechRecognition();
    recognition.lang = 'ja-JP'; // default Japanese
    recognition.interimResults = false;
    recognition.maxAlternatives = 1;

    recognition.start();

    recognition.onresult = (event) => {
      const speech = event.results[0][0].transcript;
      chatInput.value = speech;
      chatSend.click();
    };

    recognition.onerror = (event) => {
      alert("Voice recognition error: " + event.error);
    };
  });
</script>
